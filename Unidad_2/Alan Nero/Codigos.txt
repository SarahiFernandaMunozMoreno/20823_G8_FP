#include <stdio.h>
int main() {
    int vector[10];
    int i;
    for (i = 0; i < 10; i++) {
        printf("Ingrese el valor %d: ", i + 1);
        scanf("%d", &vector[i]);
    }
    for (i = 0; i < 10; i++) {
        vector[i] = vector[i] * vector[i];
    }
    printf("Vector con los cuadrados:\n");
    for (i = 0; i < 10; i++) {
        printf("%d\n", vector[i]);
    }
    return 0;
}
